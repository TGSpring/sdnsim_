136void Controller::loadTopologyFromFile(const std::string& filename) {
137 std::ifstream file(filename);
138 if (!file.is_open()) {
139 std::cerr << "Error: Could not open topology file: " << filename << std::endl;
140 return;
141 }
142
143 enum class Section { None, Devices, Links };
144 Section currentSection = Section::None;
145 /*
146 * Each line should have two device names separated by whitespace.
147 * Use std::istringstream to split.
148 */
149
150 std::string line;
151 while (std::getline(file, line)) {
152 // Trim leading whitespace
153 line.erase(0, line.find_first_not_of(" \t\r\n"));
154 // Trim trailing whitespace
155 line.erase(line.find_last_not_of(" \t\r\n") + 1);
156
157
158 if (line.empty() || line[0] == '#') {
159 //Detect section headers.
160 if (line.find("# Devices") != std::string::npos) currentSection = Section::Devices;
161 else if (line.find("# Links") != std::string::npos) currentSection = Section::Links;
162 continue;
163 }
164 std::istringstream iss(line);
165 if (currentSection == Section::Devices) {
166 std::string devName, state;
167 if (!(iss >> devName >> state)) {
168 std::cerr << "Invalid device line: " << line << std::endl;
169 continue;
170 }
171
172 addDevice(devName);
173
174 Device* dev = getDevice(devName);
175 if (dev) {
176 if (state == "active") dev->setActive(true);
177 else if (state == "inactive") dev->setActive(false);
178 else std::cerr << "Unknown state '" << state << " 'for device " << devName << std::endl;
179 }
180 }
181 else if (currentSection == Section::Links) {
182 std::string dev1, dev2;
183 if (!(iss >> dev1 >> dev2)) {
184 std::cerr << "Invalid link line: " << line << std::endl;
185 continue;
186 }
187
188 if (dev1 == dev2) {
189 std::cerr << "Warning: Ignoring self-loop for device " << dev1 << std::endl;
190 continue;
191 }
192
193 addDevice(dev1);
194 addDevice(dev2);
195
196 //Avoid duplicate connections
197 Device* d1 = getDevice(dev1);
198 if (d1) {
199 const auto& neighbors = d1->getNeighbors();
200 if (std::find(neighbors.begin(), neighbors.end(), dev2) != neighbors.end()) {
201 std::cerr << "Info: Skipping duplicate connection " << dev1 << " - " << dev2 << std::endl;
202 continue;
203 }
204 }
205 connectDevices(dev1, dev2);
206 }
207 }
208
209
210 file.close();
211 std::cout << "[INFO] Topology loaded from " << filename << std::endl;
212}
